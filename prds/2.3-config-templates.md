# PRD 2.3: Configuration Templates

## Overview
Pre-built YAML templates for common workflows.

## Deliverables
- `templates/basic-spec.yaml`
- `templates/museum-spec.yaml`
- `templates/archive-spec.yaml`
- `templates/tiff-only-spec.yaml`
- `templates/raw-only-spec.yaml`
- `src/config/templates.ts` - Template loader

## Template: basic-spec.yaml

```yaml
$schema: ./heritage-qc/schemas/config-schema.json

project:
  name: "My Project"
  # client: "Client Name"
  # date: 2025-01-19

format:
  file_type: "TIFF"
  allowed_extensions: [".tif", ".tiff"]

dimensions:
  min_long_edge: 3000

color:
  bit_depth: 16

validation:
  run_jhove: true
```

## Template: museum-spec.yaml

```yaml
$schema: ./heritage-qc/schemas/config-schema.json

project:
  name: "Museum Digitization Project"

format:
  file_type: "TIFF"
  allowed_extensions: [".tif", ".tiff"]
  compression: "Uncompressed"

dimensions:
  min_long_edge: 6000

color:
  bit_depth: 16
  color_space: "Adobe RGB (1998)"

required_exif:
  - tag: "Make"
  - tag: "Model"
  - tag: "DateTime"
  - tag: "Artist"

validation:
  run_jhove: true

naming:
  allow_spaces: false
```

## Template: raw-only-spec.yaml

```yaml
$schema: ./heritage-qc/schemas/config-schema.json

project:
  name: "RAW Camera Files"

format:
  file_type: "RAW"
  allowed_extensions: [".dng", ".arw", ".nef", ".cr2", ".cr3"]
  raw_format: "DNG"

dimensions:
  min_long_edge: 6000

required_exif:
  - tag: "Make"
  - tag: "Model"
  - tag: "LensModel"

validation:
  run_jhove: false
  check_raw_validity: true
```

## Template Loader Interface

```typescript
export type TemplateName = 'basic' | 'museum' | 'archive' | 'tiff-only' | 'raw-only';

export function listTemplates(): TemplateName[];
export function loadTemplate(name: TemplateName): Promise<ProjectSpec>;
export function getTemplatePath(name: TemplateName): string;
export function copyTemplate(name: TemplateName, destPath: string): Promise<void>;
```
